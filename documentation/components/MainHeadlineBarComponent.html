import { Component, Input } from '@angular/core';
import { CommonModule } from '@angular/common';
import { DisplayService } from '../../../services/display.service';
import { CircularButtonComponent } from '../../shared/circular-button/circular-button.component';
import { CurrentScaleService } from '../../../services/current-scale.service';
import { CurrentFretboardService } from '../../../services/current-fretboard.service';
import { RouterLink } from '@angular/router';
import { CustomizeService } from '../../../services/customize.service';
import { capitalizeFirstLetter } from '../../../utils/string.utils';

/**
 * Displays a submenu headline as an additional bar with buttons for the main submenus (scale, fretboard).
 */
@Component({
  selector: 'app-main-headline-bar',
  standalone: true,
  imports: [CommonModule, CircularButtonComponent, RouterLink],
  templateUrl: './main-headline-bar.component.html',
  styleUrl: './main-headline-bar.component.scss'
})
export class MainHeadlineBarComponent {
  @Input({ required: true }) mode!: 'scale' | 'fretboard';
  capitalize = capitalizeFirstLetter;


  /**
   * Constructor for dependency injection.
   */
  constructor(
    public currScale: CurrentScaleService,
    public currFretboard: CurrentFretboardService,
    public custom: CustomizeService,
    public display: DisplayService
  ) { }
}

<h4 [class.collapsed]="display.menuCollapsed">
    <app-circular-button *ngIf="mode === 'scale' ? currScale.isCustom : currFretboard.isCustom" [icon]="'arrow_left'"
        (click)="mode === 'scale' ? currScale.closeCustom() : currFretboard.closeCustom()" />
    {{capitalize(mode)}}
    <app-circular-button [icon]="'edit'" [info]="'Edit'" [routerLink]="custom.getEditRouteWParams(mode, true)" />
    <app-circular-button [icon]="'add'" [info]="'Add'" [routerLink]="custom.getEditRouteWParams(mode, false)" />
    <app-circular-button [icon]="'folder_open'" [info]="'Open'"
        [routerLink]="[{ outlets: { 'dialog': ['d', 'open', mode === 'scale' ? 'scale' : 'tuning'] } }]" />
</h4>
